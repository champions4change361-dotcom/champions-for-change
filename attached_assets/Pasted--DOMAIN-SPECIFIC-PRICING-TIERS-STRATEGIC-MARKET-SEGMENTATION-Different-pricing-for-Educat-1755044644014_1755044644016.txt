// 4. DOMAIN-AWARE PRICING COMPONENT
// =============================================================================

import { useDomain } from '@/hooks/useDomain';

export function DomainAwarePricing() {
  const { isSchoolDomain, isProDomain, isCoachesDomain } = useDomain();
  
  if (isSchoolDomain()) {
    return <EducationPricingSection />;
  }
  
  if (isProDomain()) {
    return <BusinessPricingSection />;
  }
  
  if (isCoachesDomain()) {
    return <CoachesLoungePricing />;
  }
  
  // Fallback
  return <EducationPricingSection />;
}
